CREATE TABLE Landkreis (
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL
)

CREATE TABLE Person(
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    vorname TEXT NOT NULL,
    nachname TEXT NOT NULL,
    jahr INTEGER NOT NULL DEFAULT 18,
    benutzername TEXT NOT NULL,
    passwort TEXT NOT NULL,
    FOREIGN KEY landkreisId REFERNECES Landkreis(id),
    fuehrerschein INTEGER NOT NULL DEFAULT 0,
    FOREIGN KEY gruppeid REFERNECES Gruppe(id)
    FOREIGN KEY eventid REFERNECES Event(id)
)

CREATE TABLE Gruppe(
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    gruppenname TEXT NOT NULL,
    status INTEGER NOT NULL DEFAULT 1
    FOREIGN KEY personid REFERNECES Person(id)
)

CREATE TABLE Event(
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    eventname TEXT NOT NULL,
    ort TEXT NOT NULL,
    zeit TEXT NOT NULL,
    bemerknung TEXT,
    FOREIGN KEY gruppeid REFERNECES Gruppe(id)
    FOREIGN KEY personid REFERNECES Person(id)
)
