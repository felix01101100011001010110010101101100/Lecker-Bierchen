CREATE TABLE Person(
    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    vorname TEXT NOT NULL,
    nachname TEXT NOT NULL,
    jahr INTEGER NOT NULL DEFAULT 18,
    benutzername TEXT NOT NULL,
    passwort TEXT NOT NULL,
    fuehrerschein INTEGER NOT NULL DEFAULT 0,
    landkreisid INTEGER NOT NULL,
    CONSTRAINT fk_Person1 FOREIGN KEY (landkreisiD) REFERENCES Landkreis(id)
